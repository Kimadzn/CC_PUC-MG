2.)
a.) x . ( x' + y' )' = x . x . y = x . ~y 

module fxyz (output s1, output s2, input  x, y, z); 
 assign s1 = x & ( ~x | ~y ); 
 assign s2 =  x & ~y;
endmodule // fxyz 

b.) ( x' + y ) + ( x . y )  = x' + ( y + x . y ) = x' + ( x.y ) = ( x' . x ) + ( x' . y ) =  1 . ( x' . y ) = x' . y

module fxyz (output s1, output s2, input  x, y, z); 
 assign s1 = ( ~x | y ) | ( x & y ); 
 assign s2 =  ~x | y; 
endmodule // fxyz 

c.) ( x' . y  )' . ( x + y' ) = ( x + y' ) . ( x + y ') = 1 

module fxyz (output s1, output s2, input  x, y, z); 
 assign s1 = ~( ~x & y  ) & ( x | ~y ); 
 assign s2 = ( x | ~y ) & ( x | ~y ); 
endmodule // fxyz 

d.) ( x' . y' )' + ( x' + y' )' = ( x + y ) + ( x . y ) = x + ( y + x . y ) = x + ( x . y ) = ( x . x ) + ( x . y )
= 1 . ( x . y ) = x | y  

module fxyz (output s1, output s2, input  x, y, z); 
 assign s1 = ~( ~x & ~y ) | ~( ~x | ~y ); 
 assign s2 =  x | y; 
endmodule // fxyz 

e.) ( y + x' )' . ( y' + x ) = ( y' . x ) . ( y' + x ) = ( y' . x + y' ) + x = ( y' . x ) + x = x . y'

module fxyz (output s1, output s2, input  x, y, z); 
 assign s1 = ~( y | ~x ) & ( ~y | x ); 
 assign s2 =  x & ~y; 
endmodule // fxyz
